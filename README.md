import numpy as np

A = np.array([[1, 2], [3, 4]])

B = np.array([[2, 0], [1, 2]])

print("element wise add:", np.add(A,B))

print("element wise sub:", np.subtract(A,B))

print("element wise mul:", np.multiply(A,B))

print("element wise div:", np.add(A,B))
